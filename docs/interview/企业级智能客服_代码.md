## 二、核心代码实现
### 1. MCP 客户端封装（用于调用订单、物流、RAG 等资源）
```java
package com.ecommerce.ai.client;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.MediaType;
import org.springframework.stereotype.Component;
import org.springframework.web.reactive.function.client.WebClient;
import reactor.core.publisher.Mono;

import java.util.Map;

/**
 * MCP 协议客户端封装
 * 用于调用 MCP 服务器暴露的 Resources（只读数据）和 Tools（有副作用操作）
 */
@Slf4j
@Component
@RequiredArgsConstructor
public class McpClient {

    private final WebClient.Builder webClientBuilder;
    private final ObjectMapper objectMapper;

    @Value("${mcp.server.url}")
    private String mcpServerUrl;

    @Value("${mcp.server.timeout}")
    private long timeout;

    /**
     * 调用 MCP Resource（只读数据，如订单、物流、RAG 检索）
     * @param uri MCP 资源 URI，如 "db://order/12345"、"rag://product/search/耳机使用说明"
     * @return 资源内容
     */
    public Mono<String> getResource(String uri) {
        log.debug("Calling MCP Resource: {}", uri);

        // 构建 MCP JSON-RPC 2.0 请求
        Map<String, Object> request = Map.of(
                "jsonrpc", "2.0",
                "id", System.currentTimeMillis(),
                "method", "resources/read",
                "params", Map.of("uri", uri)
        );

        return webClientBuilder.build()
                .post()
                .uri(mcpServerUrl)
                .contentType(MediaType.APPLICATION_JSON)
                .bodyValue(request)
                .retrieve()
                .bodyToMono(JsonNode.class)
                .map(response -> {
                    if (response.has("error")) {
                        log.error("MCP Resource error: {}", response.get("error"));
                        throw new RuntimeException("MCP Resource call failed: " + response.get("error"));
                    }
                    return response.path("result").path("content").asText();
                })
                .doOnSuccess(content -> log.debug("MCP Resource response: {}", content))
                .timeout(java.time.Duration.ofMillis(timeout));
    }

    /**
     * 调用 MCP Tool（有副作用操作，如创建工单）
     * @param toolName 工具名称
     * @param arguments 工具参数
     * @return 执行结果
     */
    public Mono<String> executeTool(String toolName, Map<String, Object> arguments) {
        log.debug("Calling MCP Tool: {}, args: {}", toolName, arguments);

        Map<String, Object> request = Map.of(
                "jsonrpc", "2.0",
                "id", System.currentTimeMillis(),
                "method", "tools/call",
                "params", Map.of(
                        "name", toolName,
                        "arguments", arguments
                )
        );

        return webClientBuilder.build()
                .post()
                .uri(mcpServerUrl)
                .contentType(MediaType.APPLICATION_JSON)
                .bodyValue(request)
                .retrieve()
                .bodyToMono(JsonNode.class)
                .map(response -> {
                    if (response.has("error")) {
                        log.error("MCP Tool error: {}", response.get("error"));
                        throw new RuntimeException("MCP Tool call failed: " + response.get("error"));
                    }
                    return response.path("result").path("content").toString();
                })
                .doOnSuccess(result -> log.debug("MCP Tool response: {}", result))
                .timeout(java.time.Duration.ofMillis(timeout));
    }
}
```

### 2. 业务逻辑层（核心流程编排）
```java
package com.ecommerce.ai.service;

import com.ecommerce.ai.client.McpClient;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.messages.SystemMessage;
import org.springframework.ai.chat.messages.UserMessage;
import org.springframework.ai.chat.prompt.Prompt;
import org.springframework.ai.vectorstore.SearchRequest;
import org.springframework.ai.vectorstore.VectorStore;
import org.springframework.stereotype.Service;
import reactor.core.publisher.Mono;
import reactor.core.scheduler.Schedulers;

import java.util.List;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
import java.util.stream.Collectors;

/**
 * 智能客服核心业务服务
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class CustomerService {

    private final ChatClient.Builder chatClientBuilder;
    private final McpClient mcpClient;
    private final VectorStore vectorStore; // Spring AI 原生向量存储（用于本地 RAG）

    // 订单号正则匹配
    private static final Pattern ORDER_PATTERN = Pattern.compile("订单[号]?[：:]?\\s*(\\d{5,10})");

    /**
     * 处理用户聊天请求
     * @param userQuery 用户输入
     * @return AI 回答
     */
    public Mono<String> chat(String userQuery) {
        log.info("Received user query: {}", userQuery);

        // 1. 解析用户意图，提取关键信息（如订单号）
        String orderId = extractOrderId(userQuery);

        // 2. 并行调用 MCP 资源和 RAG 检索（提升响应速度）
        Mono<String> orderInfoMono = orderId != null
                ? mcpClient.getResource("db://order/" + orderId)
                : Mono.just("无订单信息");

        Mono<String> logisticsInfoMono = orderId != null
                ? mcpClient.getResource("db://order/" + orderId)
                        .flatMap(order -> extractTrackingNo(order)
                                .map(trackingNo -> mcpClient.getResource("logistics://track/" + trackingNo))
                                .orElse(Mono.just("无物流信息")))
                : Mono.just("无物流信息");

        Mono<String> ragContextMono = searchProductKnowledge(userQuery);

        // 3. 组装所有上下文，调用大模型生成回答
        return Mono.zip(orderInfoMono, logisticsInfoMono, ragContextMono)
                .flatMap(tuple -> {
                    String orderInfo = tuple.getT1();
                    String logisticsInfo = tuple.getT2();
                    String ragContext = tuple.getT3();

                    String prompt = buildPrompt(userQuery, orderInfo, logisticsInfo, ragContext);
                    return callLLM(prompt);
                })
                .doOnSuccess(answer -> log.info("Generated answer: {}", answer))
                .onErrorResume(e -> {
                    log.error("Error processing query", e);
                    return Mono.just("抱歉，系统暂时无法处理您的请求，请稍后重试或联系人工客服。");
                });
    }

    /**
     * 从用户查询中提取订单号
     */
    private String extractOrderId(String query) {
        Matcher matcher = ORDER_PATTERN.matcher(query);
        return matcher.find() ? matcher.group(1) : null;
    }

    /**
     * 从订单信息中提取物流单号（简化示例）
     */
    private java.util.Optional<String> extractTrackingNo(String orderInfo) {
        // 实际生产中应解析 JSON 格式的订单信息
        return java.util.Optional.ofNullable(orderInfo.contains("trackingNo") ? "SF123456789" : null);
    }

    /**
     * 从本地向量库检索产品知识（RAG 本地实现，也可通过 MCP 调用外部 RAG）
     */
    private Mono<String> searchProductKnowledge(String query) {
        return Mono.fromCallable(() -> {
            List<org.springframework.ai.document.Document> docs = vectorStore.similaritySearch(
                    SearchRequest.query(query)
                            .withTopK(3)
                            .withSimilarityThreshold(0.7)
            );
            return docs.stream()
                    .map(org.springframework.ai.document.Document::getContent)
                    .collect(Collectors.joining("\n\n"));
        }).subscribeOn(Schedulers.boundedElastic());
    }

    /**
     * 构建 Prompt
     */
    private String buildPrompt(String userQuery, String orderInfo, String logisticsInfo, String ragContext) {
        return """
                你是某电商平台的专业智能客服，需严格遵循以下规则回答用户问题：
                1. 回答需友好、专业、简洁，避免使用技术术语；
                2. 仅基于提供的【上下文信息】回答，禁止编造内容；
                3. 若上下文信息不足，引导用户提供更多信息或转接人工客服；
                4. 涉及订单、物流信息时，需准确引用，不得出错。

                【上下文信息】
                --- 订单信息 ---
                %s

                --- 物流信息 ---
                %s

                --- 产品知识 ---
                %s

                【用户问题】
                %s
                """.formatted(orderInfo, logisticsInfo, ragContext, userQuery);
    }

    /**
     * 调用大模型
     */
    private Mono<String> callLLM(String prompt) {
        return Mono.fromCallable(() -> {
            ChatClient chatClient = chatClientBuilder.build();
            return chatClient.prompt()
                    .user(prompt)
                    .call()
                    .content();
        }).subscribeOn(Schedulers.boundedElastic());
    }
}
```

### 3. Controller 层（REST API 暴露）
```java
package com.ecommerce.ai.controller;

import com.ecommerce.ai.dto.ChatRequest;
import com.ecommerce.ai.dto.ChatResponse;
import com.ecommerce.ai.service.CustomerService;
import lombok.RequiredArgsConstructor;
import org.springframework.web.bind.annotation.*;
import reactor.core.publisher.Mono;

/**
 * 智能客服 REST API
 */
@RestController
@RequestMapping("/api/v1/customer-service")
@RequiredArgsConstructor
public class CustomerController {

    private final CustomerService customerService;

    /**
     * 聊天接口
     */
    @PostMapping("/chat")
    public Mono<ChatResponse> chat(@RequestBody ChatRequest request) {
        return customerService.chat(request.getQuery())
                .map(answer -> new ChatResponse(answer, System.currentTimeMillis()));
    }
}
```

### 5. RAG 向量库初始化（启动时加载产品文档）
```java
package com.ecommerce.ai.config;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.ai.document.Document;
import org.springframework.ai.reader.TextReader;
import org.springframework.ai.transformer.splitter.TokenTextSplitter;
import org.springframework.ai.vectorstore.VectorStore;
import org.springframework.boot.CommandLineRunner;
import org.springframework.core.io.ResourceLoader;
import org.springframework.stereotype.Component;

import java.util.List;

/**
 * RAG 向量库初始化
 * 启动时加载产品文档到 Redis 向量库
 */
@Slf4j
@Component
@RequiredArgsConstructor
public class RagInitializer implements CommandLineRunner {

    private final VectorStore vectorStore;
    private final ResourceLoader resourceLoader;

    @Override
    public void run(String... args) throws Exception {
        log.info("Initializing RAG vector store...");

        // 1. 加载产品文档（实际生产中可从文件服务器、数据库加载）
        org.springframework.core.io.Resource resource = resourceLoader.getResource("classpath:product-manuals.txt");
        TextReader reader = new TextReader(resource);

        // 2. 文档分块（TokenTextSplitter 按 token 分割，保证语义完整性）
        TokenTextSplitter splitter = new TokenTextSplitter(1000, 200);
        List<Document> documents = splitter.apply(reader.get());

        // 3. 写入向量库
        vectorStore.add(documents);

        log.info("RAG vector store initialized with {} documents", documents.size());
    }
}
```

## 三、生产环境补充建议
### 1. 安全性
- **API Key 管理**：使用 HashiCorp Vault、AWS Secrets Manager 或 Kubernetes Secrets 管理敏感配置，禁止硬编码。
- **权限控制**：集成 Spring Security，对 `/api/v1/customer-service/chat` 接口进行鉴权（如 JWT、OAuth2）。
- **数据脱敏**：在返回用户回答前，对订单号、手机号等敏感信息进行脱敏处理。

### 2. 可观测性
- **日志记录**：使用 ELK Stack（Elasticsearch + Logstash + Kibana）或 Loki 收集和分析日志。
- **监控告警**：集成 Spring Boot Actuator + Prometheus + Grafana，监控接口 QPS、P99 延迟、错误率等指标。
- **链路追踪**：集成 Micrometer Tracing + Zipkin，追踪请求在 MCP 服务器、向量库、大模型之间的调用链路。

### 3. 高可用与性能优化
- **缓存**：使用 Redis 缓存高频问题的回答（如“如何退换货”），减少大模型调用次数。
- **异步非阻塞**：全程使用 Reactor（Mono/Flux）实现异步非阻塞，提升系统吞吐量。
- **向量检索优化**：为 Redis 向量库创建 HNSW 索引，提升检索速度；对高频查询结果进行缓存。

### 4. MCP 服务器实现参考
本代码假设 MCP 服务器已存在，你可以使用 **FastMCP**（Python）或 **Spring Boot** 快速实现 MCP 服务器，封装订单库、物流 API、RAG 检索等能力。
